to delete a node at the end in a circular linked list
//
struct node*delatend(struct node*last)
{
    struct node*head=last->next;
    if(head==last)
    {
        free(last);
        return NULL;
    }
    struct node*cur=head;
    while(cur->next==last)
    {
        cur=cur->next;
    }
    cur->next=head;
    free(last);
    last=cur;
    return last;
}
