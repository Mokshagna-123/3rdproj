struct node*deletelastnode(struct node*head)
{
    if(head==NULL)
    {return NULL;
    }
    if(head->next==NULL)
    {free(head);
        return NULL;
    }
    struct node*secondlast=head;
    while(secondlast->next->next!=NULL)
    {secondlast=secondlast->next;}
    free(secondlast->next);
    secondlast->next==NULL;
    return head;
}
